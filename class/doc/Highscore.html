def inserer_score(nom_j, temp, classement)
  i = 0
  if classement
       classement.push(nom_j + " " + temp.to_s)
  else
      for joueur in  classement do
            if(joueur[1] >= temp)
                classement.insert(i, nom_j+ " " + temp.to_s)
                return
            end
            i+=1
            return if(i>=TAILLE_MAX)
      end
      classement.push(nom_j + " " + temp.to_s)
  end
end
