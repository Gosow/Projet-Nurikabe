def actualises_toi
    case(@etat.etat)
    when Etat::BLANC
        css = @@CSS_BOUTON_BLANC
        @bouton.set_label('')
    when Etat::NOIR
        css = @@CSS_BOUTON_NOIR
        @bouton.set_label('')
    when Etat::POINT
        css = @@CSS_BOUTON_BLANC
        @bouton.set_label('•')
    else
        css = @@CSS_BOUTON_NOIR
    end
    @bouton.style_context.add_provider(css, Gtk::StyleProvider::PRIORITY_USER)
end
