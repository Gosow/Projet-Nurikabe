def to_s
    (0..(@taille-1)).each do |y|
        (0..(@taille-1)).each do |x|
            print " #{
                temp = @damierCorrect[y][x]
                if temp.to_i == 0
                    temp == 'n' ? 'X' : ' '
                else
                    temp
                end
            } |"
        end
        puts ""
    end
end
